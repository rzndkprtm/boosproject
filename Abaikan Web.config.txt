SELECT
	OrderDetails.Id, 
	OrderDetails.Width,
	OrderDetails.WidthB,
	OrderDetails.[Drop],
	OrderDetails.ProductId,
	Blinds.Name AS BlindName,
	Products.Name AS ProductName
FROM
	dbo.OrderHeaders
	INNER JOIN
	dbo.OrderDetails
	ON 
		OrderHeaders.Id = OrderDetails.HeaderId
	INNER JOIN
	dbo.Products
	ON 
		OrderDetails.ProductId = Products.Id
	INNER JOIN
	dbo.ProductTubes
	ON 
		Products.TubeType = ProductTubes.Id
	INNER JOIN
	dbo.Blinds
	ON 
		Products.BlindId = Blinds.Id

	OrderHeaders.CustomerId = '928' AND
	OrderHeaders.OrderType = 'Builder' AND
	OrderHeaders.Status = 'Quoted' AND
	OrderHeaders.Active = 1 AND
	OrderDetails.Active = 1 AND
	Products.DesignId = 12 AND
	Products.TubeType = 1 AND
	Products.BlindId = 28

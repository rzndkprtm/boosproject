ALTER PROCEDURE [dbo].[sp_GetStockFabric]
    @DesignId VARCHAR(50),
    @CompanyId VARCHAR(50) = NULL,
    @Search VARCHAR(100) = NULL
AS
BEGIN
    SET NOCOUNT ON;

    -- 1. Filter Fabrics
    WITH Filtered AS (
        SELECT DISTINCT f.Id, f.Name
        FROM Fabrics f
        CROSS APPLY STRING_SPLIT(CAST(ISNULL(f.DesignId,'') AS VARCHAR(MAX)), ',') s
        CROSS APPLY STRING_SPLIT(CAST(ISNULL(f.CompanyDetailId,'') AS VARCHAR(MAX)), ',') c
        WHERE s.value = @DesignId
          AND (f.Status = 'In Stock' OR f.Status = 'Out of Stock' OR f.Status = 'Discontinued' OR f.Status = 'Limited Stock')
          
          -- ✅ Company optional
          AND (
                @CompanyId IS NULL 
                OR @CompanyId = '' 
                OR c.value = @CompanyId
              )

          -- ✅ Search optional
          AND (
                @Search IS NULL
                OR @Search = ''
                OR f.Name LIKE '%' + @Search + '%'
              )
    ),

    -- 2. Ambil colour + active
    DataCTE AS (
        SELECT 
            f.Id,
            f.Name AS FabricType,
            fc.Colour,
            fc.Status,
            ROW_NUMBER() OVER (PARTITION BY f.Id ORDER BY fc.Colour) AS rn
        FROM Filtered f
        LEFT JOIN FabricColours fc 
            ON fc.FabricId = f.Id
				WHERE fc.Status = 'In Stock' OR fc.Status = 'Out of Stock' OR fc.Status = 'Limited Stock' OR fc.Status = 'Discontinued'
    ),

    -- 3. Group per 6 kolom
    Grouped AS (
        SELECT 
            FabricType,
            Colour,
            Status,
            ((rn - 1) / 8) + 1 AS RowGroup,
            ((rn - 1) % 8) + 1 AS ColPos
        FROM DataCTE
    )

    -- 4. Pivot
    SELECT 
        FabricType,

        MAX(CASE WHEN ColPos = 1 THEN Colour END) AS Col1,
        MAX(CASE WHEN ColPos = 1 THEN Status END) AS Col1Status,

        MAX(CASE WHEN ColPos = 2 THEN Colour END) AS Col2,
        MAX(CASE WHEN ColPos = 2 THEN Status END) AS Col2Status,

        MAX(CASE WHEN ColPos = 3 THEN Colour END) AS Col3,
        MAX(CASE WHEN ColPos = 3 THEN Status END) AS Col3Status,

        MAX(CASE WHEN ColPos = 4 THEN Colour END) AS Col4,
        MAX(CASE WHEN ColPos = 4 THEN Status END) AS Col4Status,

        MAX(CASE WHEN ColPos = 5 THEN Colour END) AS Col5,
        MAX(CASE WHEN ColPos = 5 THEN Status END) AS Col5Status,

        MAX(CASE WHEN ColPos = 6 THEN Colour END) AS Col6,
        MAX(CASE WHEN ColPos = 6 THEN Status END) AS Col6Status,
				
				MAX(CASE WHEN ColPos = 7 THEN Colour END) AS Col7,
        MAX(CASE WHEN ColPos = 7 THEN Status END) AS Col7Status,
				
				MAX(CASE WHEN ColPos = 8 THEN Colour END) AS Col8,
        MAX(CASE WHEN ColPos = 8 THEN Status END) AS Col8Status

    FROM Grouped
    GROUP BY FabricType, RowGroup
    ORDER BY FabricType, RowGroup

END

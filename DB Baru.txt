INSERT INTO OrderHeaders_New SELECT Id, OrderId, CustomerId, OrderNumber, OrderName, OrderNote, OrderType, OrderFactory, Status, StatusDescription, CreatedBy, CreatedDate, QuotedDate, SubmittedDate, ProductionDate, OnHoldDate, CanceledDate, NULL, CompletedDate, NULL, NULL, NULL, NULL, NULL, NULL, NULL, NULL, 0, NULL, 0, Download, DownloadDate, Active FROM OrderHeaders

UPDATE OH
SET
OH.ShipmentDate = OS.ShipmentDate,
OH.ShipmentNumber = OS.ShipmentNumber,
OH.ContainerNumber = OS.ContainerNumber,
OH.ContainerETA = OS.ContainerETA,
OH.Courier = OS.Courier
FROM OrderHeaders_New OH
INNER JOIN OrderShipments OS
    ON OH.Id = OS.Id;

UPDATE OH
SET
OH.InvoiceDate = OI.InvoiceDate,
OH.InvoiceNumber = OI.InvoiceNumber,
OH.Collector = OI.Collector,
OH.DueDate = OI.DueDate,
OH.Payment = OI.Payment,
OH.PaymentDate = OI.PaymentDate,
OH.Amount = OI.Amount
FROM OrderHeaders_New OH
INNER JOIN OrderInvoices OI
    ON OH.Id = OI.Id;
